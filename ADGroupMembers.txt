Import-Module ActiveDirectory
$Groups = Get-ADGroup -Filter *
$Results = foreach( $Group in $Groups ){
    Get-ADGroupMember -Identity $Group | foreach {
        [pscustomobject]@{
            GroupName = $Group.Name
            Name = $_.Name
            }
        }
    }
$Results | Export-Csv .\GroupMemebers.csv -NoTypeInformation
